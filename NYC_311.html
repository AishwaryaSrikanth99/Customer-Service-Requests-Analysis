data = data.dropna(subset=['RequestClosingTime'])

def minToCategory(min):
    if (math.isnan(min)):
        return 'Unspecified'
    elif (min < 120.0):
        return 'Fast'
    elif (240.0 > min >= 120.0):
        return 'Medium'
    elif (360.0 > min >= 240.0):
        return 'Slow'
    else:
        return 'Very Slow'
